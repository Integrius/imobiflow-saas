export interface ConversionMetrics {
  // Taxa de convers√£o
  conversionRate: number; // %

  // Tempo m√©dio para convers√£o
  avgTimeToConversion: number; // dias

  // Valor m√©dio por convers√£o
  avgDealValue: number; // R$

  // Taxa de leads qualificados
  qualifiedLeadRate: number; // %

  // Score m√©dio no momento da convers√£o
  avgConversionScore: number; // 0-100
}

async function calculateConversionMetrics(
  tenantId: string,
  period: 'week' | 'month' | 'quarter'
): Promise {
  const startDate = getStartDate(period);

  const [totalLeads, conversions] = await Promise.all([
    prisma.lead.count({
      where: { tenantId, createdAt: { gte: startDate } }
    }),
    prisma.deal.findMany({
      where: {
        tenantId,
        status: 'closed_won',
        closedAt: { gte: startDate }
      },
      include: { lead: true }
    })
  ]);

  const conversionRate = (conversions.length / totalLeads) * 100;

  const avgTimeToConversion = conversions.reduce((sum, deal) => {
    const days = (deal.closedAt.getTime() - deal.lead.createdAt.getTime())
                  / (1000 * 60 * 60 * 24);
    return sum + days;
  }, 0) / conversions.length;

  const avgDealValue = conversions.reduce((sum, d) => sum + d.value, 0)
                        / conversions.length;

  const qualifiedLeads = await prisma.lead.count({
    where: { tenantId, score: { gte: 60 }, createdAt: { gte: startDate } }
  });
  const qualifiedLeadRate = (qualifiedLeads / totalLeads) * 100;

  const avgConversionScore = conversions.reduce((sum, d) => sum + d.lead.score, 0)
                              / conversions.length;

  return {
    conversionRate,
    avgTimeToConversion,
    avgDealValue,
    qualifiedLeadRate,
    avgConversionScore
  };
}

export interface AIMetrics {
  // Uso da API
  totalRequests: number;
  totalTokensUsed: number;
  avgTokensPerRequest: number;

  // Custos
  totalCost: number; // R$
  costPerLead: number; // R$
  costPerConversion: number; // R$

  // Performance
  avgResponseTime: number; // ms
  errorRate: number; // %

  // Qualidade
  avgSentimentScore: number; // -1 a 1
  accuracyScore: number; // %
}

async function calculateAIMetrics(
  tenantId: string,
  period: 'day' | 'week' | 'month'
): Promise {
  const startDate = getStartDate(period);

  // Buscar logs de API
  const apiLogs = await prisma.apiLog.findMany({
    where: {
      tenantId,
      createdAt: { gte: startDate },
      service: 'claude'
    }
  });

  const totalRequests = apiLogs.length;
  const totalTokensUsed = apiLogs.reduce((sum, log) =>
    sum + (log.inputTokens + log.outputTokens), 0
  );
  const avgTokensPerRequest = totalTokensUsed / totalRequests;

  // Calcular custos (Claude 3.5 Sonnet pricing)
  const INPUT_COST_PER_1M = 3.00;
  const OUTPUT_COST_PER_1M = 15.00;

  const totalCost = apiLogs.reduce((sum, log) => {
    const inputCost = (log.inputTokens / 1000000) * INPUT_COST_PER_1M;
    const outputCost = (log.outputTokens / 1000000) * OUTPUT_COST_PER_1M;
    return sum + inputCost + outputCost;
  }, 0);

  const leadCount = await prisma.lead.count({
    where: { tenantId, createdAt: { gte: startDate } }
  });

  const conversionCount = await prisma.deal.count({
    where: {
      tenantId,
      status: 'closed_won',
      closedAt: { gte: startDate }
    }
  });

  const costPerLead = leadCount > 0 ? totalCost / leadCount : 0;
  const costPerConversion = conversionCount > 0 ? totalCost / conversionCount : 0;

  const avgResponseTime = apiLogs.reduce((sum, log) =>
    sum + log.responseTime, 0
  ) / totalRequests;

  const errors = apiLogs.filter(log => log.error).length;
  const errorRate = (errors / totalRequests) * 100;

  return {
    totalRequests,
    totalTokensUsed,
    avgTokensPerRequest,
    totalCost,
    costPerLead,
    costPerConversion,
    avgResponseTime,
    errorRate,
    avgSentimentScore: 0.65, // Placeholder
    accuracyScore: 92 // Placeholder
  };
}

export interface EngagementMetrics {
  // Mensagens
  totalMessages: number;
  avgMessagesPerLead: number;
  avgResponseTime: number; // segundos

  // Follow-ups
  followUpsSent: number;
  followUpResponseRate: number; // %

  // Reativa√ß√£o
  coldLeadsIdentified: number;
  recoveryAttempts: number;
  recoverySuccessRate: number; // %

  // Agendamentos
  visitesScheduled: number;
  visitCompletionRate: number; // %
  noShowRate: number; // %
}

async function calculateEngagementMetrics(
  tenantId: string,
  period: string
): Promise {
  const startDate = getStartDate(period);

  const messages = await prisma.message.findMany({
    where: { tenantId, createdAt: { gte: startDate } }
  });

  const leads = await prisma.lead.findMany({
    where: { tenantId, createdAt: { gte: startDate } },
    include: { messages: true }
  });

  const followUps = await prisma.followUp.findMany({
    where: { createdAt: { gte: startDate } }
  });

  // Follow-up response rate
  const followUpResponses = await prisma.message.count({
    where: {
      tenantId,
      direction: 'inbound',
      createdAt: {
        gte: startDate,
        lte: new Date(startDate.getTime() + 24 * 60 * 60 * 1000)
      },
      leadId: { in: followUps.map(f => f.leadId) }
    }
  });

  const visits = await prisma.visit.findMany({
    where: {
      scheduledAt: { gte: startDate }
    }
  });

  const completedVisits = visits.filter(v => v.status === 'completed').length;
  const noShows = visits.filter(v => v.status === 'no_show').length;

  return {
    totalMessages: messages.length,
    avgMessagesPerLead: messages.length / leads.length,
    avgResponseTime: 45, // Placeholder - calcular real
    followUpsSent: followUps.length,
    followUpResponseRate: (followUpResponses / followUps.length) * 100,
    coldLeadsIdentified: 0, // Buscar do sistema
    recoveryAttempts: 0,
    recoverySuccessRate: 0,
    visitesScheduled: visits.length,
    visitCompletionRate: (completedVisits / visits.length) * 100,
    noShowRate: (noShows / visits.length) * 100
  };
}

export class CostTrackingService {
  async getDailyCosts(tenantId: string): Promise {
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    const apiLogs = await prisma.apiLog.findMany({
      where: {
        tenantId,
        createdAt: { gte: today }
      }
    });

    // Custos Claude
    const claudeCost = apiLogs
      .filter(log => log.service === 'claude')
      .reduce((sum, log) => {
        const inputCost = (log.inputTokens / 1000000) * 3.00;
        const outputCost = (log.outputTokens / 1000000) * 15.00;
        return sum + inputCost + outputCost;
      }, 0);

    // Custos OpenAI (se usar)
    const openaiCost = apiLogs
      .filter(log => log.service === 'openai')
      .reduce((sum, log) => {
        const inputCost = (log.inputTokens / 1000000) * 5.00;
        const outputCost = (log.outputTokens / 1000000) * 15.00;
        return sum + inputCost + outputCost;
      }, 0);

    // Outros custos fixos (hosting)
    const hostingCost = 14 / 30; // $14/m√™s = $0.47/dia

    const totalCost = claudeCost + openaiCost + hostingCost;

    // Calcular ROI
    const revenue = await this.calculateDailyRevenue(tenantId, today);
    const roi = ((revenue - totalCost) / totalCost) * 100;

    return {
      date: today,
      costs: {
        claude: claudeCost,
        openai: openaiCost,
        hosting: hostingCost,
        total: totalCost
      },
      revenue,
      roi,
      costPerLead: totalCost / (await this.getLeadCount(tenantId, today))
    };
  }

  async getMonthlyProjection(tenantId: string): Promise {
    // M√©dia dos √∫ltimos 7 dias
    const last7Days = await Promise.all(
      Array.from({ length: 7 }, (_, i) => {
        const date = new Date();
        date.setDate(date.getDate() - i);
        return this.getDailyCosts(tenantId);
      })
    );

    const avgDailyCost = last7Days.reduce((sum, day) =>
      sum + day.costs.total, 0
    ) / 7;

    return avgDailyCost * 30; // Proje√ß√£o mensal
  }

  async sendCostAlert(tenantId: string): Promise {
    const projection = await this.getMonthlyProjection(tenantId);

    if (projection > 100) { // Budget limite: $100/m√™s
      // Enviar alerta ao admin
      console.warn(`‚ö†Ô∏è Proje√ß√£o de custos: $${projection.toFixed(2)}/m√™s`);

      // Opcional: enviar email ou notifica√ß√£o
    }
  }
}

export class AlertsService {
  async checkHealthMetrics(): Promise {
    const alerts: string[] = [];

    // 1. Check error rate
    const errorRate = await this.getErrorRate();
    if (errorRate > 5) { // > 5% de erros
      alerts.push(`Taxa de erro alta: ${errorRate.toFixed(1)}%`);
    }

    // 2. Check response time
    const avgResponseTime = await this.getAvgResponseTime();
    if (avgResponseTime > 10000) { // > 10s
      alerts.push(`Tempo de resposta alto: ${avgResponseTime}ms`);
    }

    // 3. Check WhatsApp connection
    const whatsappStatus = await this.checkWhatsAppStatus();
    if (!whatsappStatus.connected) {
      alerts.push('WhatsApp desconectado!');
    }

    // 4. Check database
    const dbStatus = await this.checkDatabaseStatus();
    if (!dbStatus.healthy) {
      alerts.push('Banco de dados com problemas');
    }

    // 5. Check daily cost
    const dailyCost = await this.getDailyCost();
    if (dailyCost > 10) { // > $10/dia
      alerts.push(`Custo di√°rio alto: $${dailyCost.toFixed(2)}`);
    }

    // Enviar alertas se houver
    if (alerts.length > 0) {
      await this.sendAlerts(alerts);
    }
  }

  private async sendAlerts(alerts: string[]): Promise {
    const message =
      `üö® *ALERTAS DO SISTEMA*\n\n` +
      alerts.map((a, i) => `${i + 1}. ${a}`).join('\n');

    // Enviar para admin via Telegram
    const adminTelegramId = process.env.ADMIN_TELEGRAM_ID;
    if (adminTelegramId) {
      await telegramBot.sendMessage(adminTelegramId, message);
    }

    // Log
    console.error('üö® ALERTAS:', alerts);
  }
}
